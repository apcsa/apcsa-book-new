The code for the sumHelper method appears below. Explain in your own words why it correctly returns the sum of the integers between left and right, inclusive. 

private int sumHelper (int left, int right) {
 if (left > right) {
 return 0;
 } else {
 int mid = (left+right)/2;
 return sumHelper (left, mid-1) + mid + sumHelper (mid+1, right);
 }
}
