Again consider the buggy sumHelper. 

public int sum ( ) {
 return sumHelper (myLeft, myRight);
}

private int sumHelper (int left, int right) {
 if (left > right) {
 return 0;
 } else {
 return left + right + sumHelper (left+1, right-1);
 }
}
