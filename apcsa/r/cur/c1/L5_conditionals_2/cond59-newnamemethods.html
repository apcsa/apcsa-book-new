   public String fullName ( ) {
      if (myMiddle.equals ("")) {
         if (myLast.equals ("")) {
            return myFirst;
         }
         return myFirst + " " + myLast;
      }
      return myFirst + " " + myMiddle + " " + myLast;
   }
